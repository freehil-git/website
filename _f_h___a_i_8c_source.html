39/**
40 * @brief This function is the basic function of handling RFCommunication (<b>RobotFramework</b> <b>C</b>ommunication) for all the <b>Commands</b> related to the <b>Function</b> <b>AI</b> (Analog Input)\n
41 * This function is called from <b>FH_RFCommunication()</b>
42 *
43 * @param fh_RFCommunication_Message
44 * The message frame set by <b>RobotFramework</b>
45 *
46 * @return FH_ErrorInfo is returned
47 *
48 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
49*/
50FH_ErrorInfo FH_AI_RFCommunication(FH_RFCommunication_Message* fh_RFCommunication_Message)
51{
52 FH_ErrorInfo fh_ErrorInfo; // Error Information
53 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
54
55 uint32_t FH_AI_Data; // It temporarily stores the status of a analog input
56 uint32_t* FH_AI_Data_Ptr = &FH_AI_Data; // Pointer with &FH_AI_Data address
57
58 FH_Uint8Uint32 fh_Uint8Uint32; // To convert 32 bits data (uint32_t) to four 8 bits data (uint8_t) in FH_PWMI_Frequency
59
60 switch (fh_RFCommunication_Message -> Command)
61 {
62 case FH_AI_Commands_Init: // If the command is FH_AI_Commands_Init
63 fh_ErrorInfo = FH_AI_ProcessCommand_Init(fh_RFCommunication_Message); // Handle FH_AI_Commands_Init command
64 sprintf(FH_RFCommunication_SendBuf,"%02x|AI|Init|%02x|\n",FH_RFCommunication_DeviceAddress,fh_ErrorInfo.error_code); // set FH_RFCommunication_SendBuf buffer
65 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
66 break;
67 case FH_AI_Commands_GetStatus: // If the command is FH_AI_Commands_GetStatus
68 fh_ErrorInfo = FH_AI_ProcessCommand_GetStatus(fh_RFCommunication_Message, FH_AI_Data_Ptr); // Handle FH_AI_Commands_GetStatus command
69
70 sprintf(FH_RFCommunication_SendBuf,"%02x|AI|GetStatus|%02x|",FH_RFCommunication_DeviceAddress,fh_ErrorInfo.error_code); // set FH_RFCommunication_SendBuf buffer
71 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
72
73 fh_Uint8Uint32.value = FH_AI_Data; // set the 32 bit value
74 for (int i = 3 ; i >= 0; i--)
75 {
76 sprintf(FH_RFCommunication_SendBuf,"%02x",fh_Uint8Uint32.bytes[i]); // Format the data and set FH_RFCommunication_SendBuf buffer
77 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
78 }
79
80 sprintf(FH_RFCommunication_SendBuf,"|\n"); // set FH_RFCommunication_SendBuf buffer
81 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
82 break;
83 default:
84 break;
85 }
86 return fh_ErrorInfo;
87}

89/**
90 * @brief This function handles <b>FH_AI_Commands_Init</b> command\n
91 * There are 3 main items handled in this function as following:\n
92 * 1 - Verifying the health of the received message\n
93 * 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_AI</b> (Analog Input) to an array of function pointer\n
94 * 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_AI</b> (Analog Input) according to the first parameter of the Data in the message frame
95 *
96 * @param fh_RFCommunication_Message
97 * The message frame set by <b>RobotFramework</b>
98 *
99 * @verbatim
100 ==============================================================================
101 ##### RobotFramework Example #####
102 ==============================================================================
103 @{MessageData} = Create List ${AI01}
104 &{Message} = Create Dictionary DeviceAddress=${DeviceAdress_01} Function=${Function_AI} Command=${AI_Commands_Init} Data=@{MessageData}
105 ${Result} SendMessage &{Message}
106
107 Comments:
108
109 AI01 => It indicates the Analog Input 01
110 DeviceAdress_01 => It shall be equal to the address of the device: FH_RFCommunication_DeviceAddress
111 Function_AI => It is the function in the message frame
112 AI_Commands_Init => It is the command of the function in the message frame
113 @endverbatim
114 *
115 * @return FH_ErrorInfo is returned
116 *
117 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
118*/
119FH_ErrorInfo FH_AI_ProcessCommand_Init(FH_RFCommunication_Message* fh_RFCommunication_Message)
120{
121 FH_ErrorInfo fh_ErrorInfo; // Error Information
122 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
123
124 // Item 1 - Verifying the health of the received message
125 fh_ErrorInfo = FH_AI_Init_H(fh_RFCommunication_Message);
126 if (fh_ErrorInfo.error_code != FH_ERROR_OK) // Check if the message is healthy or not
127 {
128 return fh_ErrorInfo; // Return the error
129 }
130
131 // 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_AI</b> (Analog Input) to an array of function pointer
132 FH_AI_Init_FuncPtrModel FH_AI_Init_FuncPtr[FH_AI_MaxPeripheralNumber]; // An array of function pointer FH_AI_Init_FuncPtrModel
133 FH_AI_Init_S (FH_AI_Init_FuncPtr); // Setting the address of the functions
134
135 // 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_AI</b> (Analog Input) according to the first parameter of the Data in the message frame
136 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
137 fh_ErrorInfo = FH_AI_Init_C (fh_RFCommunication_Message, FH_AI_Init_FuncPtr); // Calling the related function according to the first parameter of the Data in the message frame
138 return fh_ErrorInfo;
139}
