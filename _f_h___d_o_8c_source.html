39/**
40 * @brief This function is the basic function of handling RFCommunication (<b>RobotFramework</b> <b>C</b>ommunication) for all the <b>Commands</b> related to the <b>Function</b> <b>DO</b> (Digital Output)\n
41 * This function is called from <b>FH_RFCommunication()</b>
42 *
43 * @param fh_RFCommunication_Message
44 * The message frame set by <b>RobotFramework</b>
45 *
46 * @return FH_ErrorInfo is returned
47 *
48 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
49*/
50FH_ErrorInfo FH_DO_RFCommunication (FH_RFCommunication_Message* fh_RFCommunication_Message)
51{
52 FH_ErrorInfo fh_ErrorInfo; // Error Information
53 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
54
55 switch (fh_RFCommunication_Message -> Command) // Check different commands
56 {
57 case FH_DO_Commands_Init: // If the command is FH_DO_Commands_Init
58 fh_ErrorInfo = FH_DO_ProcessCommand_Init(fh_RFCommunication_Message); // Handle FH_DO_Commands_Init command
59 sprintf(FH_RFCommunication_SendBuf,"%02x|DO|Init|%02x|\n",FH_RFCommunication_DeviceAddress,fh_ErrorInfo.error_code); // set FH_RFCommunication_SendBuf buffer
60 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
61 break;
62 case FH_DO_Commands_SetStatus: // If the command is FH_DO_Commands_SetStatus
63 fh_ErrorInfo = FH_DO_ProcessCommand_SetStatus(fh_RFCommunication_Message); // Handle FH_DO_Commands_SetStatus command
64 sprintf(FH_RFCommunication_SendBuf,"%02x|DO|SetStatus|%02x|\n",FH_RFCommunication_DeviceAddress,fh_ErrorInfo.error_code); // set FH_RFCommunication_SendBuf buffer
65 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
66 break;
67 case FH_DO_Commands_InitMULTI: // If the command is FH_DO_Commands_InitMULTI
68 fh_ErrorInfo = FH_DO_ProcessCommand_InitMULTI(fh_RFCommunication_Message); // Handle FH_DO_Commands_InitMULTI command
69 sprintf(FH_RFCommunication_SendBuf,"%02x|DO|InitMULTI|%02x|\n",FH_RFCommunication_DeviceAddress,fh_ErrorInfo.error_code); // set FH_RFCommunication_SendBuf buffer
70 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
71 break;
72 case FH_DO_Commands_SetStatusMULTI: // If the command is FH_DO_Commands_SetStatusMULTI
73 fh_ErrorInfo = FH_DO_ProcessCommand_SetStatusMULTI(fh_RFCommunication_Message); // Handle FH_DO_Commands_SetStatusMULTI command
74 sprintf(FH_RFCommunication_SendBuf,"%02x|DO|SetStatusMULTI|%02x|\n",FH_RFCommunication_DeviceAddress,fh_ErrorInfo.error_code); // set FH_RFCommunication_SendBuf buffer
75 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
76 break;
77 default:
78 break;
79 }
80 return fh_ErrorInfo;
81}

83/**
84 * @brief This function handles <b>FH_DO_Commands_Init</b> command\n
85 * There are 3 main items handled in this function as following:\n
86 * 1 - Verifying the health of the received message\n
87 * 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_DO</b> (Digital Output) to an array of function pointer\n
88 * 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_DO</b> (Digital Output) according to the first parameter of the Data in the message frame
89 *
90 * @param fh_RFCommunication_Message
91 * The message frame set by <b>RobotFramework</b>
92 *
93 * @verbatim
94 ==============================================================================
95 ##### RobotFramework Example #####
96 ==============================================================================
97 @{MessageData} = Create List ${DO01}
98 &{Message} = Create Dictionary DeviceAddress=${DeviceAdress_01} Function=${Function_DO} Command=${DO_Commands_Init} Data=@{MessageData}
99 ${Result} SendMessage &{Message}
100
101 Comments:
102
103 DO01 => It indicates the Digital Output 01
104 DeviceAdress_01 => It shall be equal to the address of the device: FH_RFCommunication_DeviceAddress
105 Function_DO => It is the function in the message frame
106 DO_Commands_Init => It is the command of the function in the message frame
107 @endverbatim
108 *
109 *
110 * @return FH_ErrorInfo is returned
111 *
112 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
113*/
114FH_ErrorInfo FH_DO_ProcessCommand_Init(FH_RFCommunication_Message* fh_RFCommunication_Message)
115{
116 FH_ErrorInfo fh_ErrorInfo; // Error Information
117 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
118
119 // Item 1 - Verifying the health of the received message
120 fh_ErrorInfo = FH_DO_Init_H(fh_RFCommunication_Message); // Calling the related health function
121 if (fh_ErrorInfo.error_code != FH_ERROR_OK) // Check if the message is healthy or not
122 {
123 return fh_ErrorInfo; // Return the error
124 }
125
126 // Item 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_DO</b> (Digital Output) to an array of function pointer
127 FH_DO_Init_FuncPtrModel FH_DO_Init_FuncPtr[FH_DO_MaxPeripheralNumber]; // An array of function pointer FH_DO_Init_FuncPtrModel
128 FH_DO_Init_S (FH_DO_Init_FuncPtr); // Setting the address of the functions
129
130 // Item 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_DO</b> (Digital Output) according to the first parameter of the Data in the message frame
131 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
132 fh_ErrorInfo = FH_DO_Init_C (fh_RFCommunication_Message, FH_DO_Init_FuncPtr); // Calling the related function according to the first parameter of the Data in the message frame
133 return fh_ErrorInfo;
134}
