30/**
31 * @brief This function calls the related function in <b>FH_Functions</b>, Section <b>FH_DO</b> (Digital Output) according to the first parameter of the Data in the message frame
32 *
33 * @param fh_RFCommunication_Message
34 * The message frame set by <b>RobotFramework</b>
35 * @param fH_DO_SetStatus_FuncPtr
36 * The array of function pointers
37 *
38 * @verbatim
39 ==============================================================================
40 ##### RobotFramework Example #####
41 ==============================================================================
42 @{MessageData} = Create List ${DO01} ${DO_Set}
43 &{Message} = Create Dictionary DeviceAddress=${DeviceAdress_01} Function=${Function_DO} Command=${DO_Commands_SetStatus} Data=@{MessageData}
44 ${Result} SendMessage &{Message}
45
46 Comments:
47
48 DO01 => It indicates the Digital Output 01
49 DO_Set => It indicates the Digital Output 01 shall be Set (DO_Reset indicates the Digital Output 01 shall be Reset)
50 DeviceAdress_01 => It shall be equal to the address of the device: FH_RFCommunication_DeviceAddress
51 Function_DO => It is the function in the message frame
52 DO_Commands_SetStatus => It is the command of the function in the message frame
53 @endverbatim
54 *
55 * @return FH_ErrorInfo is returned
56 *
57 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
58*/
59FH_ErrorInfo FH_DO_SetStatus_C (FH_RFCommunication_Message* fh_RFCommunication_Message, FH_DO_SetStatus_FuncPtrModel *FH_DO_SetStatus_FuncPtr)
60{
61 FH_ErrorInfo fh_ErrorInfo; // Error Information
62 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
63
64 for (uint8_t i = 1; i <= FH_DO_MaxPeripheralNumber; i++) // Iteration through all defined peripherals
65 {
66 if (i == fh_RFCommunication_Message -> Data[DO_SetStatus_PeripheralNumber]) // If the PeripheralNumber in the message frame is met
67 {
68 if (FH_DO_SetStatus_FuncPtr[i-1] != NULL_PTR) // If the address of the function is truly assigned
69 {
70 fh_ErrorInfo = FH_DO_SetStatus_FuncPtr[i-1](fh_RFCommunication_Message -> Data[1]); // Call the related function
71 }
72 else
73 {
74 fh_ErrorInfo.error_code = FH_ERROR_Root; // There is Root error
75 }
76 return fh_ErrorInfo;
77 }
78 }
79 fh_ErrorInfo.error_code = FH_ERROR_MessageFrame; // There is message frame error
80 return fh_ErrorInfo;
81}

83/**
84 * @brief This function calls the related function in <b>FH_Functions</b>, Section <b>FH_DO</b> (Digital Output) according to the 16 first parameters of the Data in the message frame\n
85 * (The first 8 bytes as mask resemble 64 bits and each bit resembles 1 <b>DO</b> (Digital Output) and if the bit is 1, the related <b>DO</b> (Digital Output) function will be called)\n
86 * (The second 8 bytes as data resemble 64 bits and each bit resembles status of 1 <b>DO</b> (Digital Output) as 1 or 0 of the corresponding bit in the first 8 bytes)
87 *
88 * @param fh_RFCommunication_Message
89 * The message frame set by <b>RobotFramework</b>
90 * @param fH_DO_SetStatus_FuncPtr
91 * The array of function pointers
92 *
93 * @verbatim
94 ==============================================================================
95 ##### RobotFramework Example #####
96 ==============================================================================
97 @{MessageData} = Create List FF FF FF FF FF FF FF FF 01 02 03 04 05 06 07 08
98 &{Message} = Create Dictionary DeviceAddress=${DeviceAdress_01} Function=${Function_DO} Command=${DO_Commands_SetStatusMULTI} Data=@{MessageData}
99 ${Result} SendMessage &{Message}
100
101 Comments:
102
103 FF FF FF FF FF FF FF FF => Example data as the first 8 parameters of the Data in the message frame utilized as mask (In this example it means all 64 digital outputs shall set status)
104 01 02 03 04 05 06 07 08 => Example data as the second 8 parameters of the Data in the message frame as data value of 1 or 0, corresponding to the first 8 bytes)
105 DeviceAdress_01 => It shall be equal to the address of the device: FH_RFCommunication_DeviceAddress
106 Function_DO => It is the function in the message frame
107 DO_Commands_SetStatusMULTI => It is the command of the function in the message frame
108 @endverbatim
109 *
110 * @return FH_ErrorInfo is returned
111 *
112 * @warning
113 * By default, in <b>SetStatusMULTI</b> (Which is mass status set of up to 64 <b>DO</b> (Digital Output)) command, 8 bytes which is 64 bits (which means 64 <b>DO</b> (Digital Output)) is used in the command\n
114 * To use <b>SetStatusMULTI</b> (Which is mass status set of up to 64 <b>DO</b> (Digital Output)) command, <b>FH_DO_MaxPeripheralNumber</b> and <b>FH_DO_MULTI_MaxPeripheralNumberLimit</b> shall be set to 64
115 *
116 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
117*/
118FH_ErrorInfo FH_DO_SetStatusMULTI_C (FH_RFCommunication_Message* fh_RFCommunication_Message, FH_DO_SetStatus_FuncPtrModel *FH_DO_SetStatus_FuncPtr)
119{
120 FH_ErrorInfo fh_ErrorInfo; // Error Information
121 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
122
123 uint8_t DO_SetStatusMULTI_StatusMaskByteCounter = 0; // It temporarily stores the index (0 - 7) of the 8 bytes of mask in the Data part of the message frame
124 uint8_t DO_SetStatusMULTI_StatusMaskByte = 0; // It temporarily stores the byte value with the index number (DO_SetStatusMULTI_StatusMaskByteCounter) of the 8 bytes of mask in the Data part of the message frame
125 uint8_t DO_SetStatusMULTI_StatusMaskBitCounter = 0; // It temporarily stores the index (0 - 7) of the every bit in the byte of mask in the Data part of the message frame
126 uint8_t DO_SetStatusMULTI_StatusMaskBit = 0; // It temporarily stores the bit value with the index number (DO_SetStatusMULTI_StatusMaskBitCounter) of every bit in the byte of mask in the Data part of the message frame
127
128 uint8_t DO_SetStatusMULTI_StatusByteCounter = 0; // It temporarily stores the index (0 - 7) of the 8 bytes of data after mask in the Data part of the message frame
129 uint8_t DO_SetStatusMULTI_StatusByte = 0; // It temporarily stores the byte value with the index number (DO_SetStatusMULTI_StatusByteCounter) of the 8 bytes of data after mask in the Data part of the message frame
130 uint8_t DO_SetStatusMULTI_StatusBitCounter = 0; // It temporarily stores the index (0 - 7) of the every bit in the byte of data after mask in the Data part of the message frame
131 uint8_t DO_SetStatusMULTI_StatusBit = 0; // It temporarily stores the bit value with the index number (DO_SetStatusMULTI_StatusBitCounter) of every bit in the byte of data after mask in the Data part of the message frame
132
133 uint8_t MaxPeripheralNumber = 0; // MaxPeripheralNumber shall be the minimum of FH_DO_MaxPeripheralNumber and FH_DO_MULTI_MaxPeripheralNumberLimit to prevent calling an undefined function (with wrong address)
134 if (FH_DO_MaxPeripheralNumber > FH_DO_MULTI_MaxPeripheralNumberLimit) // If FH_DO_MaxPeripheralNumber is bigger than FH_DO_MULTI_MaxPeripheralNumberLimit (while both shall be 64)
135 {
136 MaxPeripheralNumber = FH_DO_MULTI_MaxPeripheralNumberLimit; // The minimum is FH_DO_MULTI_MaxPeripheralNumberLimit
137 }
138 else
139 {
140 MaxPeripheralNumber = FH_DO_MaxPeripheralNumber; // The minimum is FH_DO_MaxPeripheralNumber
141 }
142
143 for (uint8_t i = 1; i <= MaxPeripheralNumber; i++) // Iteration through all defined peripherals
144 {
145 DO_SetStatusMULTI_StatusMaskByte = fh_RFCommunication_Message -> Data[DO_SetStatusMULTI_StatusMaskByteCounter]; // Temporarily store the related byte value with the related index number of the 8 bytes of mask
146 DO_SetStatusMULTI_StatusMaskBit = DO_SetStatusMULTI_StatusMaskByte & (0x80 >> DO_SetStatusMULTI_StatusMaskBitCounter); // Temporarily store the related bit value from the related byte value from the 8 bytes of mask
147
148 if (DO_SetStatusMULTI_StatusMaskBit != 0) // If the specified bit is 1, start the related process (If it is 0, it shall be neglected)
149 {
150 if (FH_DO_SetStatus_FuncPtr[i-1] != NULL_PTR) // If the address of the function is truly assigned
151 {
152 DO_SetStatusMULTI_StatusByteCounter = DO_SetStatusMULTI_StatusMaskByteCounter + 8; // 8 bytes ahead is the position of the data byte related to a mask byte
153 DO_SetStatusMULTI_StatusByte = fh_RFCommunication_Message -> Data[DO_SetStatusMULTI_StatusByteCounter]; // Temporarily store the related byte value with the related index number of the 8 bytes of data after mask
154 DO_SetStatusMULTI_StatusBit = DO_SetStatusMULTI_StatusByte & (0x80 >> DO_SetStatusMULTI_StatusBitCounter); // Temporarily store the related bit value from the related byte value from the 8 bytes of data after mask
155
156 if (DO_SetStatusMULTI_StatusBit != 0)
157 {
158 DO_SetStatusMULTI_StatusBit = 1; // Status is Set (1)
159 }
160 else
161 {
162 DO_SetStatusMULTI_StatusBit = 0; // Status is Reset (0)
163 }
164 FH_DO_SetStatus_FuncPtr[i-1](DO_SetStatusMULTI_StatusBit); // Call the related function with the specified bit value as argument
165 }
166 else
167 {
168 }
169 }
170 DO_SetStatusMULTI_StatusBitCounter++; // Increase the DO_SetStatusMULTI_StatusBitCounter
171 DO_SetStatusMULTI_StatusMaskBitCounter++; // Increase the DO_SetStatusMULTI_StatusMaskBitCounter
172 if (DO_SetStatusMULTI_StatusMaskBitCounter == 8) // If the whole bits of the byte are already processed
173 {
174 DO_SetStatusMULTI_StatusMaskBitCounter = 0; // Reset DO_SetStatusMULTI_StatusMaskBitCounter
175 DO_SetStatusMULTI_StatusMaskByteCounter++; // Increase DO_SetStatusMULTI_StatusMaskByteCounter
176 DO_SetStatusMULTI_StatusBitCounter = 0;
177 }
178 }
179 return fh_ErrorInfo;
180}
