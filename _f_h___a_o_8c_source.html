39/**
40 * @brief This function is the basic function of handling RFCommunication (<b>RobotFramework</b> <b>C</b>ommunication) for all the <b>Commands</b> related to the <b>Function</b> <b>AO</b> (Analog Output)\n
41 * This function is called from <b>FH_RFCommunication()</b>
42 *
43 * @param fh_RFCommunication_Message
44 * The message frame set by <b>RobotFramework</b>
45 *
46 * @return FH_ErrorInfo is returned
47 *
48 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
49*/
50FH_ErrorInfo FH_AO_RFCommunication(FH_RFCommunication_Message* fh_RFCommunication_Message)
51{
52 FH_ErrorInfo fh_ErrorInfo; // Error Information
53 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
54
55 switch (fh_RFCommunication_Message -> Command) // Check different commands
56 {
57 case FH_AO_Commands_Init: // If the command is FH_AO_Commands_Init
58 fh_ErrorInfo = FH_AO_ProcessCommand_Init(fh_RFCommunication_Message); // Handle FH_AO_Commands_Init command
59 sprintf(FH_RFCommunication_SendBuf,"%02x|AO|Init|%02x|\n",FH_RFCommunication_DeviceAddress,fh_ErrorInfo.error_code); // set FH_RFCommunication_SendBuf buffer
60 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
61 break;
62 case FH_AO_Commands_SetStatus: // If the command is FH_AO_Commands_SetStatus
63 fh_ErrorInfo = FH_AO_ProcessCommand_SetStatus(fh_RFCommunication_Message); // Handle FH_AO_Commands_SetStatus command
64 sprintf(FH_RFCommunication_SendBuf,"%02x|AO|SetStatus|%02x|\n",FH_RFCommunication_DeviceAddress,fh_ErrorInfo.error_code); // set FH_RFCommunication_SendBuf buffer
65 FH_RFCommunication_Send(FH_RFCommunication_SendBuf); // Send the result back to the Robot Framework
66 break;
67 default:
68 break;
69 }
70 return fh_ErrorInfo;
71}

73/**
74 * @brief This function handles <b>FH_AO_Commands_Init</b> command\n
75 * There are 3 main items handled in this function as following:\n
76 * 1 - Verifying the health of the received message\n
77 * 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_AO</b> (Analog Output) to an array of function pointer\n
78 * 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_AO</b> (Analog Output) according to the first parameter of the Data in the message frame
79 *
80 * @param fh_RFCommunication_Message
81 * The message frame set by <b>RobotFramework</b>
82 *
83 * @verbatim
84 ==============================================================================
85 ##### RobotFramework Example #####
86 ==============================================================================
87 @{MessageData} = Create List ${AO01}
88 &{Message} = Create Dictionary DeviceAddress=${DeviceAdress_01} Function=${Function_AO} Command=${AO_Commands_Init} Data=@{MessageData}
89 ${Result} SendMessage &{Message}
90
91 Comments:
92
93 AO01 => It indicates the Analog Output 01
94 DeviceAdress_01 => It shall be equal to the address of the device: FH_RFCommunication_DeviceAddress
95 Function_AO => It is the function in the message frame
96 AO_Commands_Init => It is the command of the function in the message frame
97 @endverbatim
98 *
99 * @return FH_ErrorInfo is returned
100 *
101 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
102*/
103FH_ErrorInfo FH_AO_ProcessCommand_Init(FH_RFCommunication_Message* fh_RFCommunication_Message)
104{
105 FH_ErrorInfo fh_ErrorInfo; // Error Information
106 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
107
108 // 1 - Verifying the health of the received message
109 fh_ErrorInfo = FH_AO_Init_H(fh_RFCommunication_Message);
110 if (fh_ErrorInfo.error_code != FH_ERROR_OK) // Check if the message is healthy or not
111 {
112 return fh_ErrorInfo; // Return the error
113 }
114
115 // 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_AO</b> (Analog Output) to an array of function pointer
116 FH_AO_Init_FuncPtrModel FH_AO_Init_FuncPtr[FH_AO_MaxPeripheralNumber]; // An array of function pointer FH_AO_Init_FuncPtrModel
117 FH_AO_Init_S (FH_AO_Init_FuncPtr); // Setting the address of the functions
118
119 // 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_AO</b> (Analog Output) according to the first parameter of the Data in the message frame
120 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
121 fh_ErrorInfo = FH_AO_Init_C (fh_RFCommunication_Message, FH_AO_Init_FuncPtr); // Calling the related function according to the first parameter of the Data in the message frame
122 return fh_ErrorInfo;
123}

135 *
136 * @verbatim
137 ==============================================================================
138 ##### RobotFramework Example #####
139 ==============================================================================
140 @{MessageData} = Create List ${AO01} F1 C2 01 08
141 &{Message} = Create Dictionary DeviceAddress=${DeviceAdress_01} Function=${Function_AO} Command=${AO_Commands_SetStatus} Data=@{MessageData}
142 ${Result} SendMessage &{Message}
143
144 Comments:
145
146 AO01 => It indicates the Analog Output 01
147 F1 C2 01 08 => Example data as analog value (0xF1C20108) to be set through the <b>Analog Output 01</b>\n
148 DeviceAdress_01 => It shall be equal to the address of the device: FH_RFCommunication_DeviceAddress
149 Function_AO => It is the function in the message frame
150 AO_Commands_SetStatus => It is the command of the function in the message frame
151 @endverbatim
152 *
153 *
154 * @return FH_ErrorInfo is returned
155 *
156 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
157*/
158FH_ErrorInfo FH_AO_ProcessCommand_SetStatus(FH_RFCommunication_Message* fh_RFCommunication_Message)
159{
160 FH_ErrorInfo fh_ErrorInfo; // Error Information
161 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
162
163 // 1 - Verifying the health of the received message
164 fh_ErrorInfo = FH_AO_SetStatus_H(fh_RFCommunication_Message);
165 if (fh_ErrorInfo.error_code != FH_ERROR_OK) // Check if the message is healthy or not
166 {
167 return fh_ErrorInfo; // Return the error
168 }
169
170 // 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_AO</b> (Analog Output) to an array of function pointer
171 FH_AO_SetStatus_FuncPtrModel FH_AO_SetStatus_FuncPtr[FH_AO_MaxPeripheralNumber]; // An array of function pointer FH_AO_SetStatus_FuncPtrModel
172 FH_AO_SetStatus_S (FH_AO_SetStatus_FuncPtr); // Setting the address of the functions
173
174 // 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_AO</b> (Analog Output) according to the first parameter of the Data in the message frame
175 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
176 fh_ErrorInfo = FH_AO_SetStatus_C (fh_RFCommunication_Message, FH_AO_SetStatus_FuncPtr); // Calling the related function according to the first 1 + 4 parameters of the Data in the message frame
177 return fh_ErrorInfo;
178}
