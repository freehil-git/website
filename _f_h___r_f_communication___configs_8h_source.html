33/*!
34 <b>FH_RFCommunication_MinCommandSize</b> is the minimum size of the message frame (<b>FH_RFCommunication_Message</b>) from <b>RobotFramework</b> (When DataLength is zero and there is no Data)\n
35 The minimum size includes the following items:
36 1 - DeviceAddress (1 byte)
37 2 - Function (1 byte)
38 3 - Command (1 byte)
39 4 - DataLength (1 byte) which is 0 for minimum size message
40 5 - Data (0 byte) which is not available for minimum size message
41 6 - Checksum (1 byte)
42 => Total 5 bytes of data
43*/
44#define FH_RFCommunication_MinCommandSize 5
45
46/*!
47 <b>FH_RFCommunication_ReceiveBufLength</b> is the receive buffer length of the buffer which is filled in the ISR (Interrupt Service Routine) <b>Communication Dedicated UART</b>\n
48 255 is added to FH_RFCommunication_MinCommandSize to guarantee the maximum possible Data Length of 255 bytes in the Data part of the message
49*/
50#define FH_RFCommunication_ReceiveBufLength 255 + FH_RFCommunication_MinCommandSize
51
52/*!
53 If at least one byte of data is received through the <b>ISR</b> (Interrupt Service Routine) of the <b>Communication Dedicated UART</b> by the <b>FH</b> user, a complete message shall be received in a maximum defined timeout\n
54 The received data in the <b>ISR</b> (Interrupt Service Routine) will be ignored if the complete message will not be received within the defined timeout\nFor this to happen, <b>FH_GlobalTimeCounter</b> is utilized to calculate the maximum timeout\n
55 FH_RFCommunication_ReceiveBufMaxTimeout</b> is the defined maximum timeout for every byte\n
56 In the <b>ISR</b> (Interrupt Service Routine) of the <b>Communication Dedicated UART</b> by the <b>FH</b> user, <b>FH_GlobalTimeCounter</b> shall be reset after receiving every byte of data\n
57 For example, 100 is set to have 100ms timeout (When the heart beat of the system is 1 mili-second) to receive a complete message when there is no longer any data being received
58*/
59#define FH_RFCommunication_ReceiveBufMaxTimeout 100
60
61#define FH_RFCommunication_SendBufLength 255 + FH_RFCommunication_MinCommandSize
62
63/*!
64 <b>FH_RFCommunication_Commandelements</b> includes the position of different parts of the message in the message frame (<b>FH_RFCommunication_Message</b>)
65 Since the Data part may have different sizes, the checksum position may be different
66*/
67enum FH_RFCommunication_Commandelements
68{
69 /*!
70 The 1st data in the message frame is the <b>DeviceAddress</b>\n
71 If a message (from <b>RobotFramework</b>) with a different <b>DeviceAddress</b> than the one set in <b>FH_DeviceAddress.h</b> is received, it will be ignored and not processed
72 */
73 RFCommunication_DeviceAddress = 0,
74 /*!
75 The 2nd data in the message frame is the <b>Function</b>\n
76 <b>Function</b> could be <b>DO</b> (Digital Output), <b>DI</b> (Digital Input), <b>AO</b> (Anolog Output), <b>AI</b> (Analog Input), <b>UART</b>, <b>CAN</b> , ... in <b>FH_RFCommunication_Function<b>
77 */
78 RFCommunication_Function = 1,
79 /*!
80 The 3rd data in the message frame is the <b>Command</b>\n
81 There are different commands for <b>DO</b> (Digital Output), <b>DI</b> (Digital Input), <b>AO</b> (Anolog Output), <b>AI</b> (Analog Input), <b>UART</b>, <b>CAN</b> , ...\n
82 For example for <b>DO</b> (Digital Output), different commands are available in <b>FH_DO_Configs.h</b>
83 */
84 RFCommunication_Command = 2,
85 /*!
86 The 4th data in the message frame is the <b>DataLength</b>\n
87 It is the length of the Data in the message frame\n
88 Since DataLength is just one byte, <b>Data</b> bytes can't be more than 255
89 */
90 RFCommunication_DataLength = 3,
91 /*!
92 The <b>Data</b> could be from 0 byte to up to (FH_RFCommunication_ReceiveBufLength - FH_RFCommunication_MinCommandSize) bytes\n
93 Since DataLength is just one byte, <b>Data</b> bytes can't be more than 255\n
94 <b>FH_RFCommunication_ReceiveBufLength</b> and <b>FH_RFCommunication_MinCommandSize</b> are configurable in this file (<b>FH_RFCommunication_Configs.h</b>)
95 */
96 RFCommunication_Data = 4,
97 /*!
98 The checksum of the message\n
99 */
100 // RFCommunication_Checksum = ?
101};
