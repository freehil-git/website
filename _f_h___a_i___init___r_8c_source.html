29/**
30 * @brief This function calls the related function in <b>FH_Functions</b>, Section <b>FH_AI</b> (Analog Input) according to the first parameter of the Data in the message frame
31 *
32 * @param fh_RFCommunication_Message
33 * The message frame set by <b>RobotFramework</b>
34 * @param fH_AI_Init_FuncPtr
35 * The array of function pointers
36 *
37 *
38 * @verbatim
39 ==============================================================================
40 ##### RobotFramework Example #####
41 ==============================================================================
42 @{MessageData} = Create List AI01
43 &{Message} = Create Dictionary DeviceAddress=${DeviceAdress_01} Function=${Function_AI} Command=${AI_Commands_Init} Data=@{MessageData}
44 ${Result} SendMessage &{Message}
45
46 Comments:
47
48 AI01 => It indicates the Analog Input 01
49 DeviceAdress_01 => It shall be equal to the address of the device: FH_RFCommunication_DeviceAddress
50 Function_AI => It is the function in the message frame
51 AI_Commands_Init => It is the command of the function in the message frame
52 @endverbatim
53 *
54 *
55 * @return FH_ErrorInfo is returned
56 *
57 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
58*/
59FH_ErrorInfo FH_AI_Init_R_Start (FH_RFCommunication_Message* fh_RFCommunication_Message, FH_AI_Init_FuncPtrModel *fH_AI_Init_FuncPtr)
60{
61 FH_ErrorInfo fh_ErrorInfo;
62 FH_ResetErrorInfo(&fh_ErrorInfo);
63
64 for (uint8_t i = 1; i <= FH_AI_MaxPeripheralNumber; i++)
65 {
66 if (i == fh_RFCommunication_Message -> Data[0])
67 {
68 if (fH_AI_Init_FuncPtr[i-1] != NULL_PTR)
69 {
70 fh_ErrorInfo = fH_AI_Init_FuncPtr[i-1](fh_RFCommunication_Message);
71 if (fh_ErrorInfo.error_code == FH_ERROR_NoInfo)
72 {
73 fh_ErrorInfo.error_code = FH_ERROR_Functions;
74 }
75 }
76 else
77 {
78 fh_ErrorInfo.error_code = FH_ERROR_Functions;
79 }
80 return fh_ErrorInfo;
81 }
82 }
83 fh_ErrorInfo.error_code = FH_ERROR_CommandData;
84 return fh_ErrorInfo;
85}
