200/**
201 * @brief This function handles <b>FH_UART_Commands_Init</b> command\n
202 * There are 3 main items handled in this function as following:\n
203 * 1 - Verifying the health of the received message\n
204 * 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_UART</b> to an array of function pointer\n
205 * 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_UART</b> according to the first parameter of the Data in the message frame
206 *
207 * @param fh_RFCommunication_Message
208 * The message frame set by <b>RobotFramework</b>
209 *
210 * @verbatim
211 ==============================================================================
212 ##### RobotFramework Example #####
213 ==============================================================================
214 @{MessageData} = Create List ${UART01}
215 &{Message} = Create Dictionary DeviceAddress=${DeviceAdress_01} Function=${Function_UART} Command=${UART_Commands_Init} Data=@{MessageData}
216 ${Result} SendMessage &{Message}
217
218 Comments:
219
220 UART01 => It indicates the UART 01
221 DeviceAdress_01 => It shall be equal to the address of the device: FH_RFCommunication_DeviceAddress
222 Function_UART => It is the function in the message frame
223 UART_Commands_Init => It is the command of the function in the message frame
224 @endverbatim
225 *
226 * @return FH_ErrorInfo is returned
227 *
228 * @author Vahid Hasirchi (vahid.hasirchi@gmail.com)
229*/
230FH_ErrorInfo FH_UART_ProcessCommand_Init(FH_RFCommunication_Message* fh_RFCommunication_Message)
231{
232 FH_ErrorInfo fh_ErrorInfo; // Error Information
233 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
234
235 // 1 - Verifying the health of the received message
236 fh_ErrorInfo = FH_UART_Init_H(fh_RFCommunication_Message);
237 if (fh_ErrorInfo.error_code != FH_ERROR_OK) // Check if the message is healthy or not
238 {
239 return fh_ErrorInfo; // Return the error
240 }
241
242 // 2 - Setting the addresses of the functions in <b>FH_Functions</b>, Section <b>FH_UART</b> to an array of function pointer
243 FH_UART_Init_FuncPtrModel FH_UART_Init_FuncPtr[FH_UART_MaxPeripheralNumber]; // An array of function pointer FH_UART_Init_FuncPtrModel
244 FH_UART_Init_S(FH_UART_Init_FuncPtr); // Setting the address of the functions
245
246 // 3 - Calling the related function in <b>FH_Functions</b>, Section <b>FH_UART</b> according to the first parameter of the Data in the message frame
247 FH_ResetErrorInfo(&fh_ErrorInfo); // Reset Error Information to default
248 fh_ErrorInfo = FH_UART_Init_C(fh_RFCommunication_Message, FH_UART_Init_FuncPtr); // Calling the related function according to the first parameter of the Data in the message frame
249 return fh_ErrorInfo;
250}
