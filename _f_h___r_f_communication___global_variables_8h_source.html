35/*!
36 <b>FH_RFCommunication_Message</b> is the structure for receiving the message frame from <b>RobotFramework</b>\n
37*/
38typedef struct
39{
40 /*!
41 The 1st data in the message frame is the <b>DeviceAddress</b>\n
42 If a message (from <b>RobotFramework</b>) with a different <b>DeviceAddress</b> than the one set in <b>FH_DeviceAddress.h</b> is received, it will be ignored and not processed
43 */
44 uint8_t DeviceAddress;
45 /*!
46 The 2nd data in the message frame is the <b>Function</b>\n
47 <b>Function</b> could be <b>DO</b> (Digital Output), <b>DI</b> (Digital Input), <b>AO</b> (Analog Output), <b>AI</b> (Analog Input), <b>UART</b>, <b>CAN</b> , ... in <b>FH_RFCommunication_Function<b>
48 */
49 uint8_t Function;
50 /*!
51 The 3rd data in the message frame is the <b>Command</b>\n
52 There are different commands for <b>DO</b> (Digital Output), <b>DI</b> (Digital Input), <b>AO</b> (Analog Output), <b>AI</b> (Analog Input), <b>UART</b>, <b>CAN</b> , ...\n
53 For example for <b>DO</b> (Digital Output), different commands are available in <b>FH_DO_Configs.h</b>
54 */
55 uint8_t Command;
56 /*!
57 The 4th data in the message frame is the <b>DataLength</b>\n
58 It is the length of the Data in the message frame\n
59 Since DataLength is just one byte, <b>Data</b> bytes can't be more than 255
60 */
61 uint8_t DataLength;
62 /*!
63 The <b>Data</b> could be from 0 byte to up to (FH_RFCommunication_ReceiveBufLength - FH_RFCommunication_MinCommandSize) bytes\n
64 Since DataLength is just one byte, <b>Data</b> bytes can't be more than 255\n
65 <b>FH_RFCommunication_ReceiveBufLength</b> and <b>FH_RFCommunication_MinCommandSize</b> are configurable in <b>FH_RFCommunication_Configs.h</b>
66 */
67 uint8_t Data[FH_RFCommunication_ReceiveBufLength - FH_RFCommunication_MinCommandSize];
68 /*!
69 The checksum of the message\n
70 */
71 uint8_t Checksum;
72}FH_RFCommunication_Message;
